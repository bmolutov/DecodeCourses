Links for entering the chat:
    link = bot.export_chat_invite_link(chat_id)
    # changing the old link to new variant and putting it to the variable

    # Creating new link for entering
    link = bot.create_chat_invite_link(message.chat.id, name='link1', member_limit=10)
    # Extra arguments
    # name - name of link
    # member_limit - number of users that can use this link

    # deleting link
    bot.revoke_chat_invite_link(chat_id, link)
    # example:
    link = bot.create_chat_invite_link(message.chat.id, name='link2', member_limit=10)
    # creating link
    bot.revoke_chat_invite_link(message.chat.id, link.invite_link)
    # deleting link
